DROP TABLE xsdb.connect;
CREATE TABLE IF NOT EXISTS xsdb.connect (
    sid       UB4 NOT NULL,
    connfd    UB4 NOT NULL,
    port      UB4,
    sessionid UB8X,
    hwaddr    STR(32),
    credt     STR(32),
    host      STR(127),
    ROWKEY(sid , connfd)
);


DROP TABLE xsdb.entryid;
CREATE TABLE IF NOT EXISTS xsdb.entryid (
    entryid     UB8X NOT NULL,
    status      UB4,
    mask        UB4,
    filemode    UB4,
    updtime     UB8,
    modtime     UB8,
    sessionid   UB8X,
    entry_token UB8X,
    offset_cb   UB8,
    entrykey    STR(32),
    entryfile   STR(1020),
    ROWKEY(entryid)
);


DROP TABLE xsdb.serverid;
CREATE TABLE IF NOT EXISTS xsdb.serverid(
    serverid       STR(128) NOT NULL,
    magic          UB4,
    timeout        UB4,
    sid            UB4,
    maxclients     UB4,
    stashprefix    STR(255),
    ROWKEY(serverid)
) COMMENT 'serverid table';


DROP TABLE xsdb.logentry;
CREATE TABLE IF NOT EXISTS xsdb.logentry (
    sid            UB4 NOT NULL,
    uid            UB8 NOT NULL,
    entrykey       STR(32) NOT NULL,
    status         UB4,
    addtime        UB4,
    speed_low      UB4,
    speed_high     UB4,
    updtime        UB8,
    filetime       UB8,
    entryid        UB8X,
    filesize       UB8,
    cretime        UB8,
    position       UB8,
    pathid         STR(127),
    logmd5         STR(32),
    logfile        STR(255),
    logstash       STR(1020),
    filemd5        STR(32),
    route          STR(255),
	age            UB4,
    ROWKEY(sid , uid , entrykey)
) COMMENT 'log entry table';


DROP TABLE xsdb.client;
CREATE TABLE IF NOT EXISTS xsdb.client (
    uid         UB8 NOT NULL,
    sessionid   UB8X,
    clientid    STR(127),
    ROWKEY(uid)
);


DROP TABLE xsdb.sessionid;
CREATE TABLE IF NOT EXISTS xsdb.sessionid (
    sessionid        UB8X NOT NULL,
    status           UB4,
    randnum          UB4,
    sid              UB4,
    connfd           UB4,
    uid              UB8,
    parent_sessionid UB8X,
    session_token    UB8X,
    entryid          UB8X,
    ROWKEY(sessionid)
);


DROP TABLE xsdb.clientid;
CREATE TABLE IF NOT EXISTS xsdb.clientid (
    clientid         STR(40) NOT NULL,
    status           UB4,
    maxspeed         UB4,
    maxconns         UB4,
    uid              UB8,
    cretime          UB8,
    endtime          UB8,
    secret           STR(60),
    ROWKEY(clientid)
);
